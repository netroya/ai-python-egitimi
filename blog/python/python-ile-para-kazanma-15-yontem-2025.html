🔧 Örnek Kod (Fiyat Takip Botu):


import requests
from bs4 import BeautifulSoup
import pandas as pd
from datetime import datetime

class PriceTracker:
    def __init__(self):
        self.products = []
    
    def track_product(self, url, selector):
        response = requests.get(url)
        soup = BeautifulSoup(response.content, 'html.parser')
        price = soup.select_one(selector).text.strip()
        
        return {
            'date': datetime.now(),
            'url': url,
            'price': price
        }
    
    def save_to_excel(self, filename):
        df = pd.DataFrame(self.products)
        df.to_excel(filename, index=False)

# Kullanım
tracker = PriceTracker()
# Müşteriye özel fiyat takibi
